其实也没啥区别，主要还是在应用场景上，比如：
1. 对于每个实例特有的属性（或方法）定义在this上；
2. 对于所有实例都有的方法（或属性）定义在prototype上；
3. this上一般定义属性，prototype上一般定义方法。

function Person( name, age, gender ) {

    this.name = name;
    this.age = age;
    this.gender = gender;

}

Person.prototype.eat = function ( food ) {
    // eat food
}

Person.prototype.sleep = function ( time ) {
    // sleep on time
}
