    <form action="" class="form form-horizontal">
        <div class="form-group">
            <div class="col-xs-3" id="form">
                <input type="text" class="form-control" id="formInput" value=""/>
            </div>
            <div class="col-xs-1">
                <button type="button" class="btn btn-info" id="previewBtn">preview</button>
            </div>
        </div>
        <div class="form-group">
            <div class="col-xs-6">
                <textarea class="form-control disabled" id="previewArea" disabled="disabled" rows="4"></textarea>
            </div>
        </div>
    </form>

<hr/>
<script>
    +function () {

        var previewBtn,
            previewArea,
            form,
            formInput,
            doc
            ;

        doc = window.document;

        form = doc.getElementById( "form" );
        formInput = doc.getElementById( "formInput" );
        previewBtn = doc.getElementById( "previewBtn" );
        previewArea = doc.getElementById( "previewArea" );

        previewBtn.onclick = function () {
            var html,
                    value,
                    content
                    ;
            html = form.innerHTML;
            value = formInput.value;
            content = html.replace( 'value=""', 'value="' + value + '"' );
            previewArea.value = content;
        };


    } ();
</script>
