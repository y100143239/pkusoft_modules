2. 使用

  1. 启用 Compass SCSS - <zfjd>
  2. 目录结构 
    • zfjd
    |-- config.rb
    |-- css
    |-- scss
        |-- style.scss
        |-- _login.scss
        |-- _index.scss
    |-- js
    |-- images
        |-- index
            |-- select_bg.png
            |-- select_left.png
            |-- select_right.png
  3. config.rb 
    • require 'compass/import-once/activate'
http_path = "/"
css_dir = "css"
sass_dir = "scss"
images_dir = "images"

cache = false
relative_assets = true
line_comments = false
  4. _index.scss 
    • @import "compass/utilities/sprites";

// $index-select_bg-repeat: repeat-x;    // select_bg.png 为 repeat-x

@import "index/*.png";

.select {
    @include index-sprite(select_bg);
}
.select-left {
    @include index-sprite(select_left);
}
.select-right {
    @include index-sprite(select_right);
}
